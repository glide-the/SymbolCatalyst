## 二、动态改写规则算法（伪代码）

### 算法 1：规则改写触发器

```python
function should_trigger_rule_rewrite(game_state):
    """
    判断是否应该触发规则改写
    """
    triggers = {
        'reality_event': game_state.reality_event_queue != empty,
        'character_bias': check_character_bias_trigger(game_state),
        'player_declaration': game_state.player_action == 'declare_challenge',
        'npc_intervention': check_npc_intervention(game_state),
        'pollution_level': game_state.pollution_level > 5
    }

    return any(triggers.values())
```

### 算法 2：规则改写方向计算

```python
def calculate_rewrite_direction(rule_card, trigger_source, game_state):
    """
    计算规则应该如何改写
    返回：(改写类型, 改写幅度, 目标字段)
    """

    # 改写类型枚举
    RewriteType = enum(
        'STRENGTHEN',    # 强化（效果×2）
        'WEAKEN',        # 削弱（效果÷2）
        'INVERT',        # 反转（效果变代价）
        'ADD_EXCEPTION', # 添加例外
        'COST_CHANGE'    # 代价变化
    )

    # 基于触发源计算改写方向
    if trigger_source == 'reality_event':
        # 现实事件通常导致紧急状态（强化效果，但风险上升）
        return (RewriteType.STRENGTHEN, 2.0, ['effect', 'cost'])

    elif trigger_source == 'character_bias':
        # 角色偏置触发专属改写
        character = game_state.player_character
        if character.name == '信息掮客':
            return (RewriteType.STRENGTHEN, 2.0, ['effect', 'cost'])
        elif character.name == '混沌小丑':
            return (RewriteType.COST_CHANGE, 'absurd', ['cost'])

    elif trigger_source == 'player_declaration':
        # 玩家主动挑战规则 → 进入对抗模式
        return (RewriteType.ADD_EXCEPTION, 'player_exception', ['trigger'])

    elif trigger_source == 'pollution_level':
        # 污染度高 → 规则变得不稳定（随机改写）
        return random_choice([
            (RewriteType.STRENGTHEN, random(1.5, 3.0), ['effect']),
            (RewriteType.WEAKEN, random(0.3, 0.7), ['effect']),
            (RewriteType.INVERT, None, ['effect', 'cost'])
        ])

    # 默认：不改写
    return (None, None, None)
```

### 算法 3：规则文本生成器

```python
def generate_new_rule_text(rule_card, rewrite_type, magnitude, target_fields):
    """
    基于改写方向生成新的规则文本
    """
    new_rule = rule_card.copy()

    for field in target_fields:
        if rewrite_type == 'STRENGTHEN':
            new_rule[field] = apply_multiplier(rule_card[field], magnitude)
            new_rule['description'] = f"（紧急状态）{rule_card['description']}"

        elif rewrite_type == 'WEAKEN':
            new_rule[field] = apply_multiplier(rule_card[field], magnitude)
            new_rule['description'] = f"（削弱版）{rule_card['description']}"

        elif rewrite_type == 'INVERT':
            new_rule['effect'], new_rule['cost'] = swap_fields(
                rule_card['cost'],
                rule_card['effect']
            )
            new_rule['description'] = f"（反转版）{rule_card['description']}"

        elif rewrite_type == 'ADD_EXCEPTION':
            exception_text = "玩家可以触发例外"
            new_rule['trigger'] = f"{rule_card['trigger']}，除非{exception_text}"
            new_rule['description'] = f"（玩家例外）{rule_card['description']}"

        elif rewrite_type == 'COST_CHANGE' and magnitude == 'absurd':
            # 荒诞改写（混沌小丑专属）
            new_rule['cost'] = generate_absurd_cost()
            new_rule['description'] = f"（玩笑版）{new_rule['cost']}"

    # 保留原规则作为历史版本
    new_rule['previous_version'] = rule_card
    new_rule['version'] = rule_card.get('version', 1) + 1

    # 增加污染度
    new_rule['pollution_level'] = min(10, rule_card['pollution_level'] + 1)

    return new_rule
```

### 算法 4：平衡性校验

```python
def validate_rule_balance(new_rule, game_state):
    """
    检测改写后的规则是否破坏游戏性
    返回：(是否平衡, 调整建议)
    """
    issues = []

    # 检查1：效果是否过强
    if new_rule['effect_magnitude'] > 5.0:
        issues.append("效果过强，建议生成反制卡")

    # 检查2：代价是否过低
    if new_rule['cost_magnitude'] < 0.2:
        issues.append("代价过低，建议增加代价或添加随机失败概率")

    # 检查3：污染度过高
    if new_rule['pollution_level'] > 7:
        issues.append("规则不稳定，建议添加自毁机制（3回合后恢复原版）")

    # 检查4：逻辑自洽性
    if not check_logical_consistency(new_rule):
        issues.append("逻辑矛盾，建议重新生成")

    if issues:
        return (False, issues)
    else:
        return (True, None)
```

### 算法 5：完整改写流程

```python
def rewrite_rule(rule_card, trigger_source, game_state):
    """
    完整的规则改写流程
    """
    # 步骤1：检查是否应该触发改写
    if not should_trigger_rule_rewrite(game_state):
        return rule_card

    # 步骤2：计算改写方向
    rewrite_type, magnitude, target_fields = calculate_rewrite_direction(
        rule_card,
        trigger_source,
        game_state
    )

    if rewrite_type is None:
        return rule_card

    # 步骤3：生成新规则文本
    new_rule = generate_new_rule_text(
        rule_card,
        rewrite_type,
        magnitude,
        target_fields
    )

    # 步骤4：平衡性校验
    is_balanced, issues = validate_rule_balance(new_rule, game_state)

    if not is_balanced:
        # 自动调整
        new_rule = auto_balance_rule(new_rule, issues)

        # 如果无法自动平衡，生成反制卡
        if not is_balanced:
            counter_card = generate_counter_card(new_rule)
            game_state.add_rule_card(counter_card)

    # 步骤5：应用新规则
    game_state.replace_rule_card(rule_card.id, new_rule)

    # 步骤6：增加全局污染度
    game_state.pollution_level = min(10, game_state.pollution_level + 0.5)

    # 步骤7：生成通知
    notification = generate_rule_change_notification(rule_card, new_rule)
    game_state.add_notification(notification)

    return new_rule
```
